import time
from playwright.sync_api import sync_playwright

def run(playwright):
    # Lanzar el navegador
    browser = playwright.chromium.launch(headless=False)  # headless=False para ver el navegador en acción
    # Crear una nueva página
    page = browser.new_page()
    # Navegar a Google
    page.goto('http://180.176.105.244/eut_reportes/')
    # Encontrar el campo de búsqueda y escribir una consulta
    time.sleep(2)
    page.fill('input[id="usr"]', 'C356882')
    time.sleep(2)
    # Presionar Enter para buscar
    page.fill('input[name="pass"]', 'derekNatalie72')
    time.sleep(2)
    page.press('input[name="pass"]', 'Enter')
    # Esperar a que los resultados se carguen
    time.sleep(200)
    page.wait_for_selector('h3')
    # Cerrar el navegador
    browser.close()

# Ejecutar el script con Playwright
with sync_playwright() as playwright:
    run(playwright)
