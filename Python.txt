DELETE FROM almacen_inventario 
WHERE idinventario IN (
    SELECT idinventario FROM (
        SELECT idinventario 
        FROM almacen_inventario 
        WHERE idinventario != '' 
          AND noe = '57' 
          AND almacen = 'BODEGA TI SANTA FE'
        ORDER BY idinventario ASC 
        LIMIT 160
    ) AS t
);
